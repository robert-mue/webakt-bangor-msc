1 /** Class representing a statement. 
  2 * @constructor
  3 */
  4 class Statement {
  5     /**
  6      * Create a statement.
  7      * @param {object} spec - an object-lietral containing initialising data.
  8      */
  9     constructor(spec) {
 10         this.name = spec.name;
 11         this.json = spec.json;
 12     }
 13 
 14     /**
 15      * Generates the (classic AKT) formal representation of a statement, in Prolog syntax.
 16      * @return {string} The formal version of the statement
 17      */
 18     generate_formal = function () {
 19         const infixOperator = {
 20             and:true,
 21             causes1way:true,
 22             causes2way:true,
 23             if:true,
 24             or:true};
 25             
 26         const formal = walk(this.json);
 27         this.formal = formal;
 28         return formal;
 29 
 30         function walk (part) {
 31             if (typeof part === "string") {  
 32                 return part;
 33 
 34             } else if (part.length === 3 && infixOperator[part[0]]) {
 35                 return walk(part[1])+' '+walk(part[0])+' '+walk(part[2]);
 36 
 37             } else {   // Unnecessarily generalised, since max of 3 (I think) cases.
 38                 var result = part[0]+'(';
 39                 for (var i=1; i<part.length; i++) {
 40                     var comma = i<part.length-1?',':'';
 41                     result += walk(part[i])+comma;
 42                 }
 43                 result += ')';
 44                 return result;
 45             }
 46         }
 47     }
 48 }
